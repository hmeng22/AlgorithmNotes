```cpp
1. Memoization (Top-Down) : dp table is filled on demand.
2. Tabulation (Bottom-Up) : dp table is filled one by one.

Top-Down: 一般使用迭代递归,从所需答案(Top)向下(Down)推算,并记录推算出的子问题结果
Bottom-Up: 动态规划思路,用循环完成子问题计算,并推算出最后答案
```

```cpp
Longest Increasing Subsequence (LIS)

int LIS(vector<int> &nums) {
  if(nums.empty()) return 0;
  vector<int> dp(nums.szie(), 1);
  int result = 1;
  
  for(int i = 1; i < nums.size(); i++) {
    for(int j = 0; j < i; j++) {
      if (nums[i] > nums[j]) {
        dp[i] = max(dp[i], dp[j] + 1);
        result = max(result, dp[i]);
      }
    }
  }
  
  return result;
}


vector<int> LIS_Print(vector<int> &nums) {
  vector<vector<int>> dp(nums.size());
  dp[0].push_back(nums[0]);
  vector<int> result = dp[0];
  
  for(int i = 1; i < nums.size(); i++) {
    for(int j = 0; j < i; j++) {
      if (nums[i] > nums[j] && dp[i].size() < dp[j].size() + 1) {
        dp[i] = dp[j];
        result = dp[i];
      }
    }
    
    dp[i].push_back(nums[i]);
  }
  
  return result;
}
```

```cpp
Longest Common Subsequence (LCS)

int LCS(vector<int> &a, vector<int> &b) {
  vector<vector<int>> dp(a.size() + 1, vector<int>(b.size() + 1, 0);
  
  for(int i = 1; i <= a.size(); i++) {
    for(int j = 1; j <= b.size(); j++) {
      if (a[i-1] == b[j-1]) {
        dp[i][j] = dp[i-1][j-1] + 1;
      } else {
        dp[i][j] = max(dp[i-1][j], dp[i][j-1]);
      }
    }
  }
  
  return dp[a.size()][b.size()];
}


vector<int> LCS_Print(vector<int> &a, vector<int> &b) {
  vector<vector<int>> dp(a.size() + 1, vector<int>(b.size() + 1, 0);
  
  for(int i = 1; i <= a.size(); i++) {
    for(int j = 1; j <= b.size(); j++) {
      if (a[i-1] == b[j-1]) {
        dp[i][j] = dp[i-1][j-1] + 1;
      } else {
        dp[i][j] = max(dp[i-1][j], dp[i][j-1]);
      }
    }
  }
  
  vector<int> result(dp[a.size()][b.size()]);
  
  int i = a.size();
  int j = b.size();
  while(i > 0 && j > 0) {
    if (a[i] == b[j]) {
      result.push_back(a[i]);
      i--;
      j--;
    } else if (dp[i-1][j] > dp[i][j-1]) {
      i--;
    } else {
      j--;
    }
  }
  
  reverse(result.begin(), result.end());
  
  return result;
}
```

```cpp
Edit Distance : Find minimum number of operations (Insert, Remove, Replace) to convert given string s1 into s2.

int editDistance(string s1, string s2) {
  vector<vector<int>> dp(s1.size() + 1, vector<int>(s2.size() + 1, 0));
  
  for(int i = 0; i <= s1.size(); i++) {
    dp[i][0] = i;
  }
  
  for(int j = 0; j <= s2.size(); j++) {
    dp[0][j] = j;
  }
  
  for(int i = 0; i <= s1.size(); i++) {
    for(int j = 0; j <= s2.size(); j++) {
      if (s1[i] == s2[j]) {
        dp[i][j] = dp[i-1][j-1];
      } else {
        // Insert, Remove, Replace
        dp[i][j] = 1 + min(dp[i][j-1], dp[i-1][j], dp[i-1][j-1]);
      }
    }
  }
  
  return dp[s1.size()][s2.size()];
}
```

```cpp
Minimum Cost Path : Find minimum cost of either moving down or right to the destination.

int minimumCostPath(vector<vector<int>> &cost, int row, int col) {
  vector<vector<int>> dp(row, vector<int>(col, 0));
  dp[0][0] = cost[0][0];
  
  for(int i = 1; i < row; i++) {
    dp[i][0] = dp[i-1][0] + cost[i][0];
  }
  
  for(int j = 1; j < col; j++) {
    dp[0][j] = dp[j-1][0] + cost[0][j];
  }
  
  for(int i = 1; i < row; i++) {
    for(int j = 1; j < col; j++) {
      dp[i][j] = min(dp[i-1][j], dp[i][j-1]) + cost[i][j];
    }
  }
  
  return dp[row][col];
}
```

```cpp
0-1 KnapSack

int 01KnapSack(vector<int> &weight, vector<int> &value, int W) {
  vector<vector<int>> dp(weight.size() + 1, vector<int>(W + 1, 0));
  
  for(int i = 1; i <= weight.size(); i++) {
    for(int j = 1; j <= W; j++) {
      if(weight[i] <= j) {
        dp[i][j] = max(value[i-1] + dp[i-1][j-weight[i-1]], dp[i-1][j]);
      } else {
        dp[i][j] = dp[i-1][j];
      }
    }
  }
  
  return dp[weight.size()][W];
}
```

```cpp
Longest Palindromic Subsequence

int LPC(string s) {
  vector<vector<int>> dp(s.size(), vector<int>(s.size(), 0));
  
  for(int i = 0; i < s.size(); i++) {
    dp[i][i] = 1;
  }
  
  for(int j = 0; j < s.size(); j++) {
    for(int i = j - 1; i >= 0; i--) {
      if(s[i] == s[j]) {
        dp[i][j] = dp[i+1][j-1] + 2;
      } else {
        dp[i][j] = max(dp[i][j-1], dp[i+1][j]);
      }
    }
  }
  
  return dp[0][s.size()-1];
}
```

```cpp
Longest Palindromic Substring

string LPS2(string s) {
  int maxLength = 1;
  int start = 0;
  
  for(int i = 1; i < s.size(); i++) {
    int low = i - 1;
    int high = i;
    while(low >= 0 && high < s.size() && s[low] == s[high]) {
      if(high - low + 1 > maxLength) {
        start = low;
        maxLength = high - low + 1;
      }
      --low;
      ++high;
    }
    
    low = i - 1;
    high = i + 1;
    while(low >= 0 && high < s.size() && s[low] == s[high]) {
      if(high - low + 1 > maxLength) {
        start = low;
        maxLength = high - low + 1;
      }
      --low;
      ++high;
    }
  }
  
  return s.substr(start, maxLength);
}


string LPS(string s) {
  vector<vector<int>> dp(s.size(), vector<int>(s.size(), 0));
  
  for(int i = 0; i < s.size(); i++) {
    dp[i][i] = 1;
  }
  
  int start = 0;
  int maxLength = 1;
  // check for length 2
  for(int i = 0; i < s.size() - 1; i++) {
    if(s[i] == s[i+1]) {
      dp[i][i+1] = 1;
      start = i;
      maxLength = 2;
    }
  }
  
  // check for length greater than 3
  for(int len = 3; len <= s.size(); len++) {
    for(int i = 0; i < s.size() - len + 1; i++) {
      int j = i + len - 1;
      if (dp[i+1][j-1] && s[i] == s[j]) {
        dp[i][j] = 1;
        
        if (len > maxLength) {
          start = i;
          maxLength = len;
        }
      }
    }
  }
  
  return s.substr(start, maxLength);
}
```

```cpp
Subset Sum : if a subset of the given set with sum equal to given sum.

bool subsetSum(vector<int> &nums, int target) {
  vector<vector<int>> dp(nums.size() + 1, vector<int>(target + 1, 0));
  
  for(int i = 0; i <= nums.size(); i++) {
    dp[i][0] = 1;
  }
  
  for(int j = 0; j <= target; j++) {
    dp[0][j] = 0;
  }
  
  for(int i = 1; i <= nums.size(); i++) {
    for(int j = 1; j <= target; j++) {
      if(j < nums[i]) {
        dp[i][j] = dp[i-1][j];
      }
      if(j >= nums[i]) {
        dp[i][j] = dp[i-1][j] || dp[i-1][j-nums[i-1]];
      }
    }
  }
  
  return dp[nums.size()][target];
}
```
