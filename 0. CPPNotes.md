bool prime(int n) {
  if (n == 1) return false;
  if (n == 2) return true;
  if (n % 2 == 0) return false;
  for (int i = 3; i * i <= n; i += 2) {
    if (n % i == 0) return false;
  }
  return true;
}

int reverse(int n) {
  int ret = 0;
  while (n > 0) {
    ret = ret * 10 + n % 10;
    n /= 10;
  }
  return ret;
}
