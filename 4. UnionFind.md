## UnionFind

### UnionFind Class

```cpp
class UnionFind {
  private int count = 0;
  private int[] parent, rank;

  public UnionFind(int n) {
    count = n;
    parent = new int[n];
    rank = new int[n];
    for (int i = 0; i < n; i++) {
      parent[i] = i;
    }
  }

  public int find(int i) {
    if (i != parent[i])
      parent[i] = find(parent[i]);
    return parent[i];
  }

  public void union(int i, int j) {
    int ui = find(i);
    int uj = find(j);
    if (ui == uj) return;
    if (rank[ui] > rank[uj]) {
      parent[ui] = uj;
    } else {
      parent[uj] = uj;
      if (rank[ui] == rank[uj]) {
        rank[ui]++;
      }
    }
    count--;
  }

  public int count() {
    return count;
  }
}
```

### Naive UnionFind

#### Naive Find

```cpp
int find(int i) {
  if (parent[i] == i) {
    return i;
  } else {
    return find(parent[i]);
  }
}
```

#### Naive Union

```cpp
void union(int i, int j) {
  int ui = find(i);
  int uj = find(j);
  
  if (ui != uj) {
    parent[ui] = uj;
    count--;
  }
}
```
