```cpp
bool bipartiteGraph() {
    vector<int> graph[n];
    vector<int> color(n, 0);
    
    queue<int> Q;
    for (int i = 0; i < n; i++) {
        if (color[i] != 0) continue;
        color[i] = 1;
        Q.push(i);
        while(!Q.empty()) {
            int cur = Q.front();
            Q.pop();
            for(auto next : graph[cur]) {
                if (color[next] == 0) {
                    color[next] = -1 * color[cur];
                    Q.push(next);
                } else if (color[next] != -1 * color[cur]) {
                    return false;
                }
            }
        }
    }

    return true;
}
```

### Circle in Directed Graph

```cpp
bool isDirectedGraphCyclic() {
    vector<int> graph[n];
    vector<int> inDegree(n, 0);

    for(int i = 0; i < n; i++) {
        for(int j = 0; j < graph[i].size(); j++) {
            inDegree[j]++;
        }
    }

    int count = 0;
    queue<int> Q;
    for (int i = 0; i < n; i++) {
        if (inDegree[i] == 0) {
            Q.push(i);
        }
    }
    while (!Q.empty()) {
        int cur = Q.front();
        Q.pop();
        for (auto next : graph[cur]) {
            inDegree[next]--;
            if (inDegree[next] == 0) {
                Q.push(next);
            }
        }
        graph[cur].clear();
        count++;
    }
    
    return count == n;
}
```
